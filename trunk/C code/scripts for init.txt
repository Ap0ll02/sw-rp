-- Check if this is a valid item.
		if (cider.item.stored[class]) then
			if ( hook.Call("PlayerCanDrop", GAMEMODE, player, class, true, attacker) ) then
				cider.item.make( class, player:GetPos() );
			end;
		end;
	end;
